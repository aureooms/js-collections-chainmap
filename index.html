    js-collections-chainmap

    Python's chainmap data structure for JavaScript. Parent is js-collections.

    let A = dict( [ [ "x" , "A" ] ] ) ;
let B = dict( [ [ "x" , "B" ] , [ "y" , "B" ] ] ) ;
let C = dict( [ [ "x" , "C" ] , [ "y" , "C" ] , [ "z" , "C" ] ] ) ;

let M = chainmap( A , B , C ) ;

M.get( "x" ) ; // "A"
M.get( "y" ) ; // "B"
M.get( "z" ) ; // "C"
